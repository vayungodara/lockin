-- XP drift investigation (same as 2026-03-28)
SELECT p.id, p.full_name, p.total_xp,
       COALESCE(SUM(xe.xp_gained), 0) AS events_total,
       p.total_xp - COALESCE(SUM(xe.xp_gained), 0) AS drift
FROM profiles p
LEFT JOIN xp_events xe ON xe.user_id = p.id
GROUP BY p.id, p.full_name, p.total_xp
HAVING p.total_xp != COALESCE(SUM(xe.xp_gained), 0)
   AND p.total_xp IS NOT NULL AND p.total_xp != 0
ORDER BY drift DESC;
