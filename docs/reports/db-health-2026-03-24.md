-- Fix XP drift: reconcile profiles.total_xp with actual xp_events sum
-- Option A: Reset profile XP to match events (loses unlogged XP)

SET total_xp = sub.computed_xp
FROM (
  SELECT p2.id, COALESCE(SUM(xe.xp_gained), 0) AS computed_xp
  FROM public.profiles p2
  LEFT JOIN public.xp_events xe ON xe.user_id = p2.id

) sub
WHERE p.id = sub.id
  AND p.total_xp != sub.computed_xp

FROM public.profiles p
LEFT JOIN public.xp_events xe ON xe.user_id = p.id
GROUP BY p.id, p.total_xp
