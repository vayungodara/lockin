**Fix XP drift — reconcile profiles.total_xp with xp_events sum:**

```sql
-- Option A: Reset profile XP to match xp_events (loses the "extra" XP)
UPDATE public.profiles p
SET total_xp = COALESCE(sub.computed_xp, 0)
FROM (
  SELECT xe.user_id, SUM(xe.xp_gained) AS computed_xp
  FROM public.xp_events xe
  GROUP BY xe.user_id
) sub
WHERE p.id = sub.user_id
  AND p.total_xp != sub.computed_xp;

-- Option B: Insert missing xp_events to match profile totals (preserves current XP)
-- Requires deciding what event_type to assign. Manual review recommended.

-- Option C: Add a DB trigger to keep them in sync going forward
CREATE OR REPLACE FUNCTION sync_profile_xp()
RETURNS TRIGGER AS $$
BEGIN
  UPDATE public.profiles
  SET total_xp = (
    SELECT COALESCE(SUM(xp_gained), 0)
    FROM public.xp_events
    WHERE user_id = NEW.user_id
  )
  WHERE id = NEW.user_id;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_sync_xp
AFTER INSERT OR UPDATE OR DELETE ON public.xp_events
FOR EACH ROW EXECUTE FUNCTION sync_profile_xp();
```
